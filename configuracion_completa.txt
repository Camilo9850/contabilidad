SOLUCIÓN INTEGRAL PARA EL PROBLEMA DE ELIMINACIÓN

1. ACTUALIZA EL CONTROLADOR (FacturacionController.php):

public function eliminar(Request $request)
{
    $id = $request->input('id');
    
    $facturacion = Facturacion::find($id);
    if (!$facturacion) {
        if ($request->ajax() || $request->wantsJson()) {
            return response()->json(['err' => '1', 'msg' => 'Facturación no encontrada'], 404);
        }
        return redirect()->route('facturacion.index')->with('error', 'Facturación no encontrada');
    }

    try {
        $facturacion->delete();
        if ($request->ajax() || $request->wantsJson()) {
            return response()->json(['err' => '0', 'msg' => 'Registro eliminado exitosamente.']);
        } else {
            return redirect()->route('facturacion.index')->with('success', 'Facturación eliminada correctamente');
        }
    } catch (\Exception $e) {
        if ($request->ajax() || $request->wantsJson()) {
            return response()->json(['err' => '1', 'msg' => 'Error al eliminar el registro: ' . $e->getMessage()], 500);
        } else {
            return redirect()->route('facturacion.index')->with('error', 'Error al eliminar la facturación');
        }
    }
}
